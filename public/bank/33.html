SubObjective:

Declare and use variables of primitive data types

In the script of the following HTML page, there are functions that operate on strings:

<!DOCTYPE html>
<html>
<head>
      <title>Intro to JavaScript</title>
           <meta charset="utf-8" />
</head>
<body>
      <script>
           var lorem = `Lorem Ipsum is simply dummy text of the printing and
                               typesetting industry.Lorem Ipsum has been the industry's
                               standard dummy text ever since the 1500s, when an unknown
                               printer took a galley of type and scrambled it to make a type
                               specimen book.It has survived not only five centuries, but
                               also the leap into electronic typesetting, remaining
                               essentially unchanged.`;

           function func1() {
                var i1 = 6;
                var i2 = 11;
                var result = lorem.substring(i1, i2).toUpperCase();
                console.log(result);
           }

           function func2() {
                var i = lorem.indexOf("essentially");
                var j = i + 12;
                var newString = lorem.slice(j, j + 10).repeat(3);
                console.log(newString);
           }

           function func3() {
                var newString = lorem.substring(0, 27).trim().split(' ');
                console.log(newString.length);
           }

           function func4() {
                var i = lorem.indexOf(" ' ");
                var newString = lorem.substring(0, i);
                var result;
                if (newString.endsWith('1500'))
                     result += newString.toUpperCase();
                else
                     result += newString.lastIndexOf('industry');
                console.log(result);
           }

           func1();
           func2();
           func3();
           func4();

      </script>
</body>
</html>

What would these functions return? To answer, select the appropriate options from the drop-down menus.

Explanation

The variable lorem contains a multi-line text using the back-tick string declaration. Each function does some operations on the string lorem and displays a result in the console.

func1() declares two indexes and uses the substring function to read a splice from the lorem string. substring() takes two arguments for the start and the end. It starts at 6 and ends at 11. The result is then converted to uppercase and printed in the console. Looking at the indexes in the lorem string, it is clear that the string Ipsum is returned from the substring and converted to uppercase, resulting in "IPSUM".

func2() looks for the index of "essentially", adding 12 to that index, slicing the string lorem from that index to 10 additional characters, resulting in "unchanged.". This slice is then repeated three times to give "unchanged. unchanged. unchanged.".

func3() reads a substring from 0 to 27, trimming any spaces resulting in "Lorem Ipsum is simply dummy". Then this text is split using a space. This creates an array of 5 elements "Lorem", "Ipsum", "is", "simply", and "dummy". Finally, the length of that array is printed, which is 5.

func4() looks for the index of the single quote ("'"), reads a substring of lorem from the beginning up to that index, then tests whether the resulting string ends with "1500". Since it does not, the last index of "industry" is added to result and printed in the console. This results is 111. However, since result was never initialized, adding a number to undefined results in NaN, not 111.
